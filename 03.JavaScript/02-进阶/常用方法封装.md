```javascript
function arrayUnique(arr) {
  const obj = {}
  return arr.filter(item => {
    if(!obj[item]) {
      obj[item] = true
      return true
    }
  })
}

// 2.3
function arrayUnique(arr) {
  const result = []
  arr.forEach((item) => {
    if(result.indexOf(item) === -1) {
      result.push(item)
    }
  })
  return result
}
```

9.  字符串去重

```javascript
// 3.1 
String.prototype.unique = function () {
  let result = ''
  const obj = {}
  const len = this.length
  for (let i = 0; i < len; i++) {
    if (!obj[this[i]]) {
      result += this[i]
      obj[this[i]] = true
    }
  }
  return result
}

// 3.2 去除连续的字符串
function strUnique(str) {
  return str.replace(/(\w)\1+/g, '$1')
}
```
